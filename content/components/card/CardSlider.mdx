Serves as the footer for the Card component.

## Usage

```javascript
import {
  CardSlider
} from "thousand-lights";

const CardComponent = () => {

    const cardItems = [
        {
          title: 'Card 1 ',
          description: 'Lorem ipsum dolor sit amet et nuncat mergitur',
          buttonTitle: 'Get Started',
          backgroundColor: 'black',
          buttonFunction: function(){
              console.log("Thousand Lights")
          }
        },
        {
            title: 'Card 2',
            description: 'Lorem ipsum dolor sit amet et nuncat mergitur',
            buttonTitle: 'Get Started',
            backgroundColor: 'black',
            buttonFunction: ()=>{AlertFunction()}
          },
      ];


  return (
    <CardSlider data={cardItems} />
  )
}

```
