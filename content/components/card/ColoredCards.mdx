List of colored Cards that can be configured to fit your design.

## Usage

```javascript
import ColoredCards from "thousand-lights";

const ColoredCards = () => {
     const cardItems = [
        {
          title: 'Card 1 ',
          description: 'Lorem ipsum dolor sit amet et nuncat mergitur',
          buttonTitle: 'Get Started',
          backgroundColor: 'black',
          buttonFunction: function(){
              console.log("Thousand Lights")
          }
        },
        {
            title: 'Card 2',
            description: 'Lorem ipsum dolor sit amet et nuncat mergitur',
            buttonTitle: 'Get Started',
            backgroundColor: 'black',
            buttonFunction: ()=>{AlertFunction()}
          },
      ];
  return (
    <View style={{flex:1}}
    <ColoredCards data={cardItems} />
    </View>
  );
};

```
